#include "380.txt"

int main()
{
	if (getAccess(getPlayer()) == 1)
		{
		string Msg;
		Msg = getName(getPlayer());
		Msg += " ^RIS CHEATING BY USING AN ADMIN ROUND TABLE ITEM.";
		sendChat(Msg, -1);
		return 0;
		}

	int PlayerID;
	int MapID;

	PlayerID = getPlayer();
	MapID = getMap(PlayerID);

	MergeSubmap(&TABLE_ROUND, MapID, getX(PlayerID), getY(PlayerID));
	updateMap(MapID);
}
