// Targeted Heal Spell

int main()
{
if (getAccess(getPlayer()) == 1)
     {
     string Msg;
     Msg = getName(getPlayer());
     Msg += " ^RIS CHEATING BY USING AN ADMIN HEAL ITEM.";
     sendChat(Msg, -1);
     return 0;
     }
	int id=getPlayer();
	int cost=25;
	int map=getMap(id);
	int x  =getX(id);
	int y  =getY(id);
	int victim=getMTarget(id);
	if(getMP(id)<cost)
	{
		sendChat("^RNot enough mana.",id);
	}
	else if(victim>-1)
	{		
		setMP(id,getMP(id)-cost);
		makeEffect(map,getX(victim),getY(victim),1);
		setHP(victim,getHP(victim)+magicDamage(id));
	}
}
