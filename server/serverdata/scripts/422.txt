// BBQ Meat

void main()
{
	int id=getPlayer();
	int cmeat=itemSlot(id,360);
	int sherb=itemSlot(id,514);
	if(cmeat>-1 || sherb>-1)
	{
		int mqty=qtyAt(id,cmeat);
		int hqty=qtyAt(id,sherb);
		if(mqty>=1 || hqty>=1)
		{
			setQty(id,cmeat,mqty-1);
			setQty(id,sherb,hqty-1);
			giveItem(id,513,1);
		}
		else
		{
			sendChat("^RNot enough ingredients!",id);
		}
	}
	else
	{
		sendChat("^RNot enough ingredients!",id);
	}
}
