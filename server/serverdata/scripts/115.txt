void main()
{
	int PlayerID;
	int TargetID;
	string TargetName;
	int InvSize;
	int i;
	int ItemID;
	int Quantity;
	string Msg;
	int Length;

	PlayerID = getPlayer();
	TargetID = getMTarget(PlayerID);

	if (TargetID == -1)
		{
		sendChat("^RError: No target selected", PlayerID);
		return;
		}

	TargetName = getName(TargetID);
	InvSize = inventoryLen();

	Msg = "^GListing ";
	Msg += TargetName;
	Msg += "'s Inventory:";
	sendChat(Msg, PlayerID);

	for (i = 0; i < InvSize; i++)
		{
		ItemID = itemAt(TargetID, i);
		Quantity = qtyAt(TargetID, i);

		if (Quantity == 0)
			continue;

		Msg = "^YSlot #";
		Msg += itos(i);
		Msg += ": ";
		Msg += getItemName(ItemID);

		if (Quantity > 1)
			{
			Msg += "[";
			Msg += itos(Quantity);
			Msg += "]";
			}

		sendChat(Msg, PlayerID);
		}
}
