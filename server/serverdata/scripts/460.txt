// Elementalist Purge

void main()
{
	int id=getPlayer();
	int map=getMap(id);
	int x  =getX(id);
	int y  =getY(id);
	if(getHP(id)<=15)
	{
		sendChat("^RToo weak to purge.",id);
	}
	else
	{
		makeEffect(map,x,y,17);
		setHP(id,getHP(id)-getRand(5)-10);
		setMP(id,getMP(id)+getRand(40)+20);
	}
}
