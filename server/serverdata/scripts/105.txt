const int TURKEY_SPAWN_SCRIPT_ID = 100;

void main()
{
	int PlayerID;
	int TimerID;

	PlayerID = getPlayer();

	sendChat("^YTurkeys will now begin spawning", PlayerID);

	TimerID = runScript(TURKEY_SPAWN_SCRIPT_ID, 1);

	setGlobalInt("TurkeyTimerID", TimerID);
}
