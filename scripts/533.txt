#include"mapping.h"

int iteration=0;
int mapid=STARTMAP;
int mapy=0;
void main()
{

	for(int mapx=0;mapx<NUMMAPS;mapx++)
	{
		cMap *map=mapStruct(mapid);
		clear(mapid);
		for(int x=0;x<20;x++)
		{
			for(int y=0;y<15;y++)
			{
				if((mapx==0&&x<WATER_BORDER)||(mapx==NUMMAPS-1&&x>20-WATER_BORDER)||(mapy==0&&y<WATER_BORDER)||(mapy==NUMMAPS-1&&y>15-WATER_BORDER))
				{
					map->map[y][x].tile[0]=WATER_TILE;
					map->map[y][x].type=TILE_TYPE_WALL;
				}
				else if(!iteration)
				{
					map->map[y][x].tile[0]=GRASS_TILE;
					map->map[y][x].type=TILE_TYPE_NONE;
				}
				else
				{
					map->map[y][x].tile[0]=DIRT_TILE;
					map->map[y][x].type=TILE_TYPE_WALL;
				}
				
				if(iteration)
				{
					map->map[y][x].tile[3]=SHADOW_TILE;
				}
			}
		}
		if(!iteration)
		{
			sprintf(map->name,"New Mystera %d %d",mapx,mapy);
			map->npc[4].index=40;
		}
		else
		{
			sprintf(map->name,"The Underground %d %d",mapx,mapy);
			map->npc[4].index=25;
			map->npc[4].qty=-60;
		}
		
		if(mapy!=0)
		{
			map->n=mapid-NUMMAPS;
		}
		else
		{
			map->n=mapid;
		}
		
		if(mapy<NUMMAPS - 1)
		{
			map->s=mapid+NUMMAPS;
		}
		else
		{
			map->s=mapid;
		}
		
		if(mapx!=0)
		{
			map->w=mapid-1;
		}
		else
		{
			map->w=mapid;
		}
		
		if(mapx<NUMMAPS - 1)
		{
			map->e=mapid+1;
		}
		else
		{
			map->e=mapid;
		}
		printf ("%d %d %d populated\n",mapx,mapy, mapid);
		mapid++;
	}
	
	if(mapy<NUMMAPS - 1)
	{
		mapy++;
		runScript(myScript(),1);
	}
	else if(iteration==0)
	{
		mapy=0;
		iteration=1;
		runScript(myScript(),1);
	}
	else
	{
		mapy=0;
		iteration=0;
		mapid=STARTMAP;
		runScript(534,1);
	}
}
