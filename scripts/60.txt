//House Tile Identifier
#include "HOChecker.h"

void main()
{
	int PlayerID = getPlayer();
	if(!houseRunner())
	{
		sendChat("^rYour not in your house.",PlayerID);
		return;
	}
	int map,x,y;
	map=getMap(PlayerID);
	x=getX(PlayerID);
	y=getY(PlayerID);
	if(getDir(PlayerID)==0)y++;
	if(getDir(PlayerID)==1)y--;
	if(getDir(PlayerID)==2)x--;
	if(getDir(PlayerID)==3)x++;

	makeEffect(map,x,y,19);


		
	int founditem=1000;
	int fi;
	int img;
	
	if(getTileType(map,x,y)==1)
	{
		sendChatf(PlayerID,"Empty");
		return;
	}
	if(getTileType(map,x,y)==2)
	{
		sendChatf(PlayerID,"Wall");
		return;
	}
	if(getTileType(map,x,y)==3)
	{
		sendChatf(PlayerID,"Warp");
		return;
	}
	if(getTileType(map,x,y)==4)
	{
		sendChatf(PlayerID,"NPC");
		return;
	}
	if(getTileType(map,x,y)==5)
	{
		sendChatf(PlayerID,"Script");
		return;
	}
	if(getTileType(map,x,y)==6)
	{
		sendChatf(PlayerID,"Switch");
		return;
	}
	if(getTileType(map,x,y)==7)
	{
		sendChatf(PlayerID,"Keep");
		return;
	}
	

}
