//Shovel
#include"mapping.h"
#include"skills.h"
#include"items.h"

int findshovel(int id)
{
	int shovel;
	for(int i=0;i<26;i++)
	{
		shovel=itemAt(id,i);
		if(shovel==STONE_SHOVEL ||shovel==COPPER_SHOVEL||shovel==IRON_SHOVEL)
		{
			return shovel;
		}
	}
	return 0;	
}
void main()
{
	int id=getPlayer();
	cPlayer *player=playerStruct(id);
	int x=player->x;
	int y=player->y;
	int mapid=player->map;
	int dir=player->dir;
	cMap *map=mapStruct(mapid);
	int skillevel=getSkill(id,SHOVEL);
	int getexp=0;
	int shovel=findshovel(id);
	int basechance=5;
	
	if(skillevel==-1)
	{
		setSkill(id,SHOVEL,1);
		skillevel=1;
	}
	
	if(shovel==STONE_SHOVEL)
	{
		basechance=15+skillevel;
	}
	else if(shovel==COPPER_SHOVEL)
	{
		basechance=25+skillevel;
	}
	else if(shovel==IRON_SHOVEL)
	{
		basechance=35+skillevel;
	}
	
	if(dir==DIR_UP)
		y--;
	else if(dir==DIR_DOWN)
		y++;
	else if(dir==DIR_LEFT)
		x--;
	else if(dir==DIR_RIGHT)
		x++;
		
	if(!inBounds(x,y))
		return;
		
	if(mapid<1200)
	{
		cMap *undermap=mapStruct(mapid+1200);
		bool wateround=tilearound(map,x,y,WATER_TILE);
		if(map->map[y][x].type==TILE_TYPE_NONE&&map->map[y][x].tile[1]==0 && !wateround)
		{
			if(undermap->map[y][x].tile[0]!=WATER_TILE && getRand(100)<basechance/2&& x>0 && y>0 && y<14 && x<19)
			{
				map->map[y][x].type=TILE_TYPE_SCRIPT;
				map->map[y][x].warp.map=8;
				map->map[y][x].tile[1]=STAIR_DOWN;
				undermap->map[y][x].tile[1]=STAIR_UP;
				undermap->map[y][x].type=TILE_TYPE_SCRIPT;
				undermap->map[y][x].warp.map=8;
				undermap->map[y][x].tile[0]=DUG_DIRT;
				savemap(mapid+1200);
				savemap(mapid);
				getexp+=getRand(10)+10;
			}
			else if(getRand(100)<basechance/2)
			{
				map->map[y][x].type=TILE_TYPE_WALL;
				map->map[y][x].tile[0]=WATER_TILE;
				smootharound(mapid,map,x,y);
				savemap(mapid);
				sendChat("^BWater bubbles up from the depths!",id);
				getexp+=getRand(15)+10;
			}
		}
		else if(wateround && map->map[y][x].tile[0]!=WATER_TILE && getRand(100)<basechance)
		{
			if(x==0||y==0||x==19||y==14)
			{
				extendtonext(map,x,y,WATER_TILE,TILE_TYPE_WALL,TILE_TYPE_NONE);
				getexp+=getRand(5)+2;
			}
			if(map->map[y][x].type==TILE_TYPE_NONE&&map->map[y][x].tile[1]==0)
			{
				map->map[y][x].tile[0]=WATER_TILE;
				map->map[y][x].type=TILE_TYPE_WALL;
				smootharound(mapid,map,x,y);
				savemap(mapid);
				getexp+=getRand(5)+2;
			}
		}
	}
	else if(mapid<2400)
	{
		cMap *overmap=mapStruct(mapid-1200);
		bool wateround=tilearound(map,x,y,WATER_TILE);
		if(map->map[y][x].type==TILE_TYPE_WALL&&getRand(100)<basechance)
		{
			if(map->map[y][x].tile[1]==0)
			{
				if(x==0||y==0||x==19||y==14)
				{
					extendtonext(map,x,y,DUG_DIRT,TILE_TYPE_NONE,TILE_TYPE_WALL);
					getexp+=getRand(5)+2;
				}
				map->map[y][x].type=TILE_TYPE_NONE;
				map->map[y][x].tile[0]=DUG_DIRT;
				savemap(mapid);
				getexp+=getRand(10)+5;
			}
		}
		else if(wateround && map->map[y][x].tile[0]==DUG_DIRT &&  getRand(100)<basechance)
		{
			map->map[y][x].tile[0]=WATER_TILE;
			map->map[y][x].type=TILE_TYPE_WALL;
			smootharound(mapid,map,x,y);
			savemap(mapid);
			getexp+=getRand(10)+5;
		}
		else if(map->map[y][x].type==TILE_TYPE_NONE && getRand(100)<basechance/3 && x>0 && y>0 && y<14 && x<19)
		{
				map->map[y][x].type=TILE_TYPE_SCRIPT;
				map->map[y][x].warp.map=8;
				map->map[y][x].tile[1]=STAIR_UP;
				overmap->map[y][x].tile[1]=STAIR_DOWN;
				overmap->map[y][x].type=TILE_TYPE_SCRIPT;
				overmap->map[y][x].warp.map=8;
				savemap(mapid-1200);
				savemap(mapid);
				getexp+=getRand(5)+5;
		}
	}
	
	if(getexp!=0)
	{
		makeItem(7,1,mapid,player->x,player->y);
		int exp=getSkill(id,SHOVEL_EXP)+getexp;
		setSkill(id,SHOVEL_EXP,exp);
		if(exp>levelUpExp(skillevel))
		{
			setSkill(id,SHOVEL,skillevel+1);
			sendChatf(id,"^GYou have reached the digging level of %d!",skillevel+1);
		}
	}
}
