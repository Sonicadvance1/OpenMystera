// Herbal Meat

void main()
{
	int id=getPlayer();
	int item1=itemSlot(id,3);
	int item2=itemSlot(id,5);
	if(item1>-1 && item2>-1)
	{
		int qty1=qtyAt(id,item1);
		int qty2=qtyAt(id,item2);
		if(qty1>=1 && qty2>=2)
		{
			setQty(id,item1,qty1-1);
			setQty(id,item2,qty2-2);
			giveItem(id,17,1);
		}
		else
		{
			sendChat("^RYou do not have the required ingredients.",id);
		}
	}
	else
	{
		sendChat("^RYou do not have the required ingredients.",id);
	}
}
