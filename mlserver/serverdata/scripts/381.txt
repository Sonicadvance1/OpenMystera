// Death Touch

void main()
{
	int id=getPlayer();
	int map=getMap(id);
	int x  =getX(id);
	int y  =getY(id);
	switch(getDir(id))
	{
		case 0:
			y++;
			makeEffect(map,x,y,21);
			break;
		case 1:
			y--;
			makeEffect(map,x,y,23);
			break;
		case 2:
			x--;
			makeEffect(map,x,y,22);
			break;
		case 3:
			x++;
			makeEffect(map,x,y,20);
			break;
	}
	int victim=getID(map,x,y);
	setHP(victim,0);
}
